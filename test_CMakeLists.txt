cmake_minimum_required(VERSION 3.21)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)

find_package(Qt6    REQUIRED COMPONENTS Core)
find_package(GTest  REQUIRED)
find_package(Boost  REQUIRED COMPONENTS system)
find_package(OpenSSL REQUIRED)

enable_testing()
include(GoogleTest)

add_executable(EventTest
    EventTest.cpp
    ../event.cpp
)
target_include_directories(EventTest PRIVATE ..)
target_link_libraries(EventTest PRIVATE GTest::gtest_main Qt6::Core)
gtest_discover_tests(EventTest)

add_executable(AuthManagerTest
    AuthManagerTest.cpp
)
target_include_directories(AuthManagerTest PRIVATE ..)
target_link_libraries(AuthManagerTest PRIVATE GTest::gtest_main Qt6::Core)
gtest_discover_tests(AuthManagerTest)

add_executable(NetworkManagerTest
    NetworkManagerTest.cpp
    ../networkmanager.cpp
)
target_include_directories(NetworkManagerTest PRIVATE ..)
target_link_libraries(NetworkManagerTest PRIVATE
    GTest::gmock_main
    Qt6::Core
    Boost::system
    OpenSSL::SSL
    OpenSSL::Crypto
)
gtest_discover_tests(NetworkManagerTest)
